        QtGui.QWidget.__init__(self, parent)
        self.setupUi(self)

        self.commands = commands
        self.channels = channels
        self.applicationSettings = applicationSettings
        self.projectSettings = projectSettings
        self.communicator = communicator

        self.movePlot = True
