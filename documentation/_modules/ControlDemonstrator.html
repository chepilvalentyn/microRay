[docs]    def notify(self, object, event):
        isex = False
        try:
            return QtGui.QApplication.notify(self, object, event)
        except:
            isex = True
            logging.error("uncaught exception:")
            logging.error(traceback.format_exc())
            return False
        finally:
            if isex:
                self.quit()
